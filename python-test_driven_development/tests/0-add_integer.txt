#!/usr/bin/python3
"""
Unit tests for add_integer function.
"""
import unittest
from 0-add_integer import add_integer


class TestAddInteger(unittest.TestCase):

    def test_positive_integers(self):
        self.assertEqual(add_integer(1, 2), 3)

    def test_negative_integers(self):
        self.assertEqual(add_integer(-1, -2), -3)

    def test_zero(self):
        self.assertEqual(add_integer(0, 0), 0)

    def test_float_inputs(self):
        self.assertEqual(add_integer(1.5, 2.5), 3)
        self.assertEqual(add_integer(-1.5, -2.5), -3)

    def test_float_and_integer(self):
        self.assertEqual(add_integer(1, 2.9), 3)
        self.assertEqual(add_integer(2.9, 1), 3)

    def test_default_argument(self):
        self.assertEqual(add_integer(2), 100)

    def test_invalid_a_type(self):
        with self.assertRaises(TypeError):
            add_integer("1", 2)

    def test_invalid_b_type(self):
        with self.assertRaises(TypeError):
            add_integer(1, "2")

    def test_large_numbers(self):
        self.assertEqual(add_integer(1000000000, 1000000000), 2000000000)


if __name__ == '__main__':
    unittest.main()
